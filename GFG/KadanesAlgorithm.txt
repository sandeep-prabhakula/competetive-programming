class Solution{

    // arr: input array
    // n: size of array
    //Function to find the sum of contiguous subarray with maximum sum.
    long maxSubarraySum(int arr[], int n){
        
        long sum = Integer.MIN_VALUE;
        long curSum = 0;
        for(int i:arr){
            curSum += i;
            if(curSum>sum)sum = curSum;
            if(curSum<0)curSum = 0;
        }
        return sum;
    }
    
}
