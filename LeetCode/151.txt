class Solution {
    public String reverseWords(String s) {
        String [] str = s.split(" ");
        String res = "";
        for(int i=str.length-1;i>=0;i--){
            if(str[i].length()>0){
                res+=str[i]+" ";
            }
        }
        return res.length()>0?res.substring(0, res.length()-1):res;
    }
}
