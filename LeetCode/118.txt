//pascalTriangle
class Solution {
    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>>res = new ArrayList<>();
        for(int i=1;i<=numRows;i++){
            int c = 1;
            List<Integer>list = new ArrayList<>();
            for(int j=1;j<=i;j++){
                list.add(c);
                c = c*(i-j)/j;
            }
            res.add(list);
        }
        return res;
    }
}
